    PEO --> PLO
    PLO --> CLO
    CLO --> CO

    PLO --> IA

    Quiz & Assign & Mid & Final & Lab & Proj --> SCA
    CES --> SCA
    SCA --> CCA
    SCA --> SPA
    GES & AS & ES --> SPA
    SPA --> PPA
    
    CCA --> GAP
    PPA --> GAP
    GAP --> AP
    AP --> APO
    APO --> RC
